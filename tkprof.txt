// Ver plan de ejecución
SELECT LPAD(' ', 2*LEVEL) || OPERATION || ' '
       || OPTIONS || ' ' || OBJECT_NAME || ' '
       || OBJECT_TYPE ||  ' ' || OBJECT_INSTANCE || ' '
       || OBJECT_OWNER || ' ' || OPTIMIZER || ' '
       || POSITION || ' ' || ID || ' '
       || PARENT_ID || ' '
       AS query_plan
FROM PLAN_TABLE
WHERE STATEMENT_ID = 'P1'
CONNECT BY PRIOR ID = PARENT_ID
START WITH ID = 0;
